# ======================
# 1. IMPORTURI

import os
import cv2
import numpy as np
import tensorflow as tf

from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import (
    Conv2D,
    MaxPooling2D,
    Flatten,
    Dense,
    Dropout
)
# ======================
# 2. EXTRAGERE CADRE
# ======================

def extragere_cadre(video_path, nr_cadre=25):

    cap = cv2.VideoCapture(video_path)

    total_cadre = int(
        cap.get(cv2.CAP_PROP_FRAME_COUNT)
    )

    cadre=[]

    indexi=np.linspace(
        0,
        total_cadre-1,
        nr_cadre
    ).astype(int)


    for index in indexi:

        cap.set(
            cv2.CAP_PROP_POS_FRAMES,
            index
        )

        ret,cadru=cap.read()

        if ret:

            cadru=cv2.cvtColor(
                cadru,
                cv2.COLOR_BGR2RGB
            )

            cadru=cv2.resize(
                cadru,
                (224,224)
            )

            cadre.append(cadru)


    cap.release()

    return np.array(cadre)
# ======================
# 3. INCARCARE DATASET
# ======================

def incarca_dataset(folder):

    X=[]
    y=[]


    real=os.path.join(folder,"real")
    fake=os.path.join(folder,"fake")


    for video in os.listdir(real):

        if video.endswith(".mp4"):

            cadre=extragere_cadre(
                os.path.join(real,video)
            )

            for cadru in cadre:

                X.append(cadru)
                y.append(0)



    for video in os.listdir(fake):

        if video.endswith(".mp4"):

            cadre=extragere_cadre(
                os.path.join(fake,video)
            )

            for cadru in cadre:

                X.append(cadru)
                y.append(1)



    return np.array(X), np.array(y)
# ======================
# 4. MODEL CNN
# ======================

def creare_model():

    model=Sequential([

        Conv2D(
            32,
            (3,3),
            activation="relu",
            input_shape=(224,224,3)
        ),

        MaxPooling2D(),

        Conv2D(
            64,
            (3,3),
            activation="relu"
        ),

        MaxPooling2D(),

        Flatten(),

        Dense(
            128,
            activation="relu"
        ),

        Dropout(0.5),

        Dense(
            1,
            activation="sigmoid"
        )
    ])


    model.compile(
        optimizer="adam",
        loss="binary_crossentropy",
        metrics=["accuracy"]
    )

    return model
